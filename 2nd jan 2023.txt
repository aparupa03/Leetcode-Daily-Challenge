class Solution {
public:
    bool detectCapitalUse(string word) {
        for(int i = 0; i < word.size(); i++){
            if(i == 1 && word[i-1] > 96 && word[i] < 91){
                return false;
            }
            if(i > 1 && ((word[i]>96 && word[i-1]<91) || (word[i]<91 && word[i-1]>96))){
                return false;
            }
        }
        return true;
    }
};
