//N-th Tribonacci Number
class Solution {
public:
    int tribonacci(int n) {
        if(n == 0) return 0;
        if(n == 1 || n == 2) return 1;
        int t0 = 0, t1 = 1, t2 = 1;
        int t3 = t2 + t1 + t0;
        for(int i = 4; i <= n; i++){
            t0 = t1;
            t1 = t2;
            t2 = t3;
            t3 = t0 + t1 + t2; 
        }
        return t3;
    }
};
